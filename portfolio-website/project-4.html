reward_calculator.py
def calculate_reward(prediction, reference):
    # Calculate semantic similarity using CLIP
    similarity = clip_model.encode(prediction) @ clip_model.encode(reference)
    
    # Check for geometric artifacts
    integrity_score = mesh_evaluator.check_manifold(prediction.mesh)
    
    if integrity_score < 0.5:
        return -1.0 # Penalize broken geometry
        
    # Weighted reward sum
    total_reward = (similarity * 0.7) + (integrity_score * 0.3)
    
    return max(0.0, total_reward)
